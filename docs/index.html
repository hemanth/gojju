Code Examples

Functional Pipeline
# Pipe operator for elegant data transformations
[1, 2, 3, 4, 5]
  |> filter(\x -> x % 2 == 0)
  |> map(\x -> x * 2)
  |> sum
# => 12
Pattern Matching
fn describe(n)
  match n
    0 -> "zero"
    n if n < 0 -> "negative"
    n if n < 10 -> "small"
    _ -> "large"
  end
end

describe(5)  # => "small"
Functions & Lambdas
# Regular function
fn greet(name)
  "Hello, #{name}!"
end

# Arrow function

# Haskell-style lambda
